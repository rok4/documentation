63  Attribute(json11::Json doc) {
64 
65  missingField = "";
66  values = std::vector<std::string>();
67  min = "";
68  max = "";
69  metadataJson = "";
70 
71  if (! doc["name"].is_string()) {
72  missingField = "name";
73  return;
74  }
75  name = doc["name"].string_value();
76 
77  if (! doc["type"].is_string()) {
78  missingField = "type";
79  return;
80  }
81  type = doc["type"].string_value();
82 
83  if (! doc["count"].is_number()) {
84  missingField = "count";
85  return;
86  }
87  count = std::to_string(doc["count"].number_value());
88 
89  if (doc["min"].is_number()) {
90  min = std::to_string(doc["min"].number_value());
91  }
92  if (doc["max"].is_number()) {
93  max = std::to_string(doc["max"].number_value());
94  }
95  if (doc["values"].is_array()) {
96  std::string tmp;
97  for (json11::Json v : doc["values"].array_items()) {
98  tmp = v.string_value();
99  boost::replace_all(tmp, "\"", "\\\"");
100  values.push_back(tmp);
101  }
102 
103  }
104 
105  };
106  ~Attribute(){};
107

109  std::string getName() {return name;}
110  std::string getType() {return type;}
111  std::vector<std::string> getValues() {return values;}
112  std::string getCount() {return count;}
113  std::string getMin() {return min;}
114  std::string getMax() {return max;}

116  std::string getMetadataJson() {
117  if (metadataJson != "") return metadataJson;
118  /*
119  {
120  "attribute": "gid",
121  "count": 1,
122  "max": 49,
123  "min": 49,
124  "type": "number",
125  "values": [
126  49

128  }
129  */
130 
131  std::ostringstream res;
132  res << "{\"attribute\":\"" << name << "\"";
133  res << ",\"count\":" << count << "";
134  res << ",\"type\":\"" << type << "\"";
135 
136  if (min != "") {
137  res << ",\"min\":" << min;
138  }
139  if (max != "") {
140  res << ",\"max\":" << max;
141  }
142 
143  if (values.size() != 0) {
144  res << ",\"values\":[\"" << boost::algorithm::join(values, "\",\"") << "\"]";
145  }
146 
147  res << "}";
148 
149  metadataJson = res.str();
150  return metadataJson;
151  }
