324 static bool readHeaderFile(std::string imagefilename, int* width, int* height, int* samplesperpixel, int* bitspersample) {
325 
326  const char * pch;
327  pch = strrchr ( imagefilename.c_str(),'.' );
328  int basenamelength = pch - imagefilename.c_str();
329 
330  char headerfilename[IMAGE_MAX_FILENAME_LENGTH];
331  memset(headerfilename, 0, IMAGE_MAX_FILENAME_LENGTH);
332  memmove(headerfilename, imagefilename.c_str(), basenamelength);
333 
334  memcpy(headerfilename+basenamelength, ".hdr", 4);
335 
336  std::ifstream header;
337 
338  // Tentative d'ouverture du fichier header, avec l'extension en minuscule ou en majuscule
339  header.open ( headerfilename );
340  if ( !header ) {
341  memcpy(headerfilename+basenamelength, ".HDR", 4);
342  header.open ( headerfilename );
343  if ( !header ) {
344  BOOST_LOG_TRIVIAL(error) << "No found header, with extension .hdr or .HDR" ;
345  return false;
346  }
347  }
348 
349  std::string str;
350  char key[20];
351  int value = 0;
352  while (! header.eof() ) {
353  std::getline ( header,str );
354 
355  if ( std::sscanf ( str.c_str(),"%s %d", key, &value) != 2 ) continue;
356 
357  if (strncmp(key, "NCOLS", 5) == 0 || strncmp(key, "ncols", 5) == 0) *width = value;
358  if (strncmp(key, "NROWS", 5) == 0 || strncmp(key, "nrows", 5) == 0) *height = value;
359  if (strncmp(key, "NBITS", 5) == 0 || strncmp(key, "nbits", 5) == 0) *bitspersample = value;
360  if (strncmp(key, "NBANDS", 6) == 0 || strncmp(key, "nbands", 5) == 0) *samplesperpixel = value;
361  }
362 
363  return true;
364 }

380 static bool uncompressedData(uint8_t* compresseddata, int compresseddatasize, uint8_t* rawdata, int rawdatasize) {
381 
382  // Initialisation du flux
383  z_stream zstream;
384  zstream.zalloc = Z_NULL;
385  zstream.zfree = Z_NULL;
386  zstream.opaque = Z_NULL;
387  zstream.data_type = Z_BINARY;
388  int zinit;
389  if ( ( zinit=inflateInit ( &zstream ) ) != Z_OK ) {
390  if ( zinit==Z_MEM_ERROR ) BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : pas assez de memoire" ;
391  else if ( zinit==Z_VERSION_ERROR ) BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : versions de zlib incompatibles" ;
392  else if ( zinit==Z_STREAM_ERROR ) BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : parametres invalides" ;
393  else BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : echec" ;
394  return false;
395  }
396 
397  zstream.next_in = compresseddata;
398  zstream.avail_in = compresseddatasize;
399  zstream.next_out = rawdata;
400  zstream.avail_out = rawdatasize;
401  // Decompression du flux
402  while ( zstream.avail_in != 0 ) {
403  if ( int err = inflate ( &zstream, Z_SYNC_FLUSH ) ) {
404  if ( err == Z_STREAM_END && zstream.avail_in == 0 ) break; // fin du fichier OK.
405  if ( zstream.avail_out == 0 ) {
406  BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : le buffer de sortie ne devrait pas être trop petit";
407  return false;
408  }
409  BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : probleme deflate decompression " << err ;
410  return false;
411  }
412  }
413 
414  // Destruction du flux
415  if ( inflateEnd ( &zstream ) != Z_OK ) {
416  BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : probleme de liberation du flux" ;
417  return false;
418  }
419 
420  if (zstream.avail_out != 0) {
421  BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : la taille des données décompressée n'est pas celle attendue" ;
422  BOOST_LOG_TRIVIAL(error) << "Il manque " << zstream.avail_out << "octets" ;
423  return false;
424  }
425 
426  return true;
427 }
