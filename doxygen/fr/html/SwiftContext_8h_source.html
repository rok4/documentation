253  void closeConnection() {

256  if(token_file != "" && ! use_token_from_file) {

259  if ( token_stream.is_open() ) {
260  token_stream << token;
261  token_stream.close();
262  BOOST_LOG_TRIVIAL(debug) << "Token file " << token_file << " updated with new token " << token ;
263  } else {
264  BOOST_LOG_TRIVIAL(warning) << "Token file " << token_file << " could not be updated with new token " << token;
265  }
266  }
267  else if(token_file != "") {
268  BOOST_LOG_TRIVIAL(debug) << "Token file " << token_file << " already contains the used token " << token ;
269  }
270  }
