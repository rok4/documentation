171  template<typename T>
172  void unassociateAlpha ( T* buffer ) {
173 
174  T coeff;
175  if (sizeof(T) == 1) {
176  // pour les entiers sur 8 bits
177  coeff = (T) 255;
178  } else if (sizeof(T) == 2) {
179  // pour les entiers sur 16 bits
180  coeff = (T) 65535;
181  } else {
182  coeff = (T) 1;
183  }
184 
185  T* pix = buffer;
186  int alphaInd = channels - 1;
187  for (int i = 0; i < width; i++, pix += channels) {
188  T alpha = *(pix + alphaInd);
189 
190  if (alpha == coeff) {
191  // Opacité pleine
192  continue;
193  }
194  if (alpha == 0) {
195  // Transparence complète
196  memset(pix, 0, channels);
197  continue;
198  }
199 
200  for (int c = 0; c < alphaInd; c++) {
201  pix[c] = pix[c] * coeff / alpha;
202  }
203  }
204  }
