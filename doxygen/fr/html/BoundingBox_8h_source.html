252  void phase(BoundingBox<T> other, T resx, T resy) {
253 
254  double intpart;
255  double phi = 0;
256  double phaseDiff = 0;
257 
258  // phase de la bbox fournie, de référence, sur laquelle se caser
259  double phaseX = other.getPhaseX(resx);
260  double phaseY = other.getPhaseY(resy);
261 
262  // Mise en phase de xmin (sans que celui ci puisse être plus petit)
263  phi = getPhaseX(resx);
264  if ( phi < 0. ) {
265  phi += 1.0;
266  }
267 
268  if ( fabs ( phi-phaseX ) > 0.0001 && fabs ( phi-phaseX ) < 0.9999 ) {
269  phaseDiff = phaseX - phi;
270  if ( phaseDiff < 0. ) {
271  phaseDiff += 1.0;
272  }
273  xmin += phaseDiff * resx;
274  }
275 
276  // Mise en phase de xmax (sans que celui ci puisse être plus grand)
277  phi = getPhaseX(resx);
278  if ( phi < 0. ) {
279  phi += 1.0;
280  }
281 
282  if ( fabs ( phi-phaseX ) > 0.0001 && fabs ( phi-phaseX ) < 0.9999 ) {
283  phaseDiff = phaseX - phi;
284  if ( phaseDiff > 0. ) {
285  phaseDiff -= 1.0;
286  }
287  xmax += phaseDiff*resx;
288  }
289 
290  // Mise en phase de ymin (sans que celui ci puisse être plus petit)
291  phi = getPhaseY(resy);
292  if ( phi < 0. ) {
293  phi += 1.0;
294  }
295 
296  if ( fabs ( phi-phaseY ) > 0.0001 && fabs ( phi-phaseY ) < 0.9999 ) {
297  phaseDiff = phaseY - phi;
298  if ( phaseDiff < 0. ) {
299  phaseDiff += 1.0;
300  }
301  ymin += phaseDiff*resy;
302  }
303 
304  // Mise en phase de ymax (sans que celui ci puisse être plus grand)
305  phi = getPhaseY(resy);
