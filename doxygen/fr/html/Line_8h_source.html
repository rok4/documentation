297 template<typename T>
298 void Line::write ( T* buffer, int outChannels ) {
299  switch ( outChannels ) {
300  case 1:
301  for ( int i = 0; i < width; i++ ) {
302  buffer[i] = ( T ) ( 0.2125*samples[3*i] + 0.7154*samples[3*i+1] + 0.0721*samples[3*i+2] ) * alpha[i];
303  }
304  break;
305  case 2:
306  for ( int i = 0; i < width; i++ ) {
307  buffer[2*i] = ( T ) ( 0.2125*samples[3*i] + 0.7154*samples[3*i+1] + 0.0721*samples[3*i+2] );
308  buffer[2*i+1] = ( T ) ( alpha[i]*coeff );
309  }
310  break;
311  case 3:
312  for ( int i = 0; i < width; i++ ) {
313  buffer[3*i] = ( T ) (alpha[i] * samples[3*i]);
314  buffer[3*i+1] = ( T ) (alpha[i] * samples[3*i+1]);
315  buffer[3*i+2] = ( T ) (alpha[i] * samples[3*i+2]);
316  }
317  break;
318  case 4:
319  for ( int i = 0; i < width; i++ ) {
320  buffer[4*i] = ( T ) (samples[3*i]);
321  buffer[4*i+1] = ( T ) (samples[3*i+1]);
322  buffer[4*i+2] = ( T ) (samples[3*i+2]);
323  buffer[4*i+3] = ( T ) ( alpha[i]*coeff );
324  }
325  break;
326  }
327 }
