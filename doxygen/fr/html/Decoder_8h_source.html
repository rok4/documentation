83 class DataSourceDecoder : public DataSource {
84 private:
85  DataSource* encData;
86  const uint8_t* decData;
87  size_t decSize;
88 public:
89  DataSourceDecoder ( DataSource* encData ) : encData ( encData ), decData ( 0 ), decSize ( 0 ) {}
90 
91  ~DataSourceDecoder() {
92  if ( decData )
93  delete[] decData;
94  delete encData;
95  }
96 
97  const uint8_t* getData ( size_t &size ) {
98  if ( !decData && encData ) {
99  decData = Decoder::decode ( encData, decSize );
100  if ( !decData ) {
101  delete encData;
102  encData = 0;
103  }
104  }
105  size = decSize;
106  return decData;
107  }
108 
109  bool releaseData() {
110  if ( encData ) encData->releaseData();
111  if ( decData ) delete[] decData;
112  decData = 0;
113  return true;
114  }

120  return 200;
121  }
122  std::string getEncoding() {
123  return "";
124  }

126  unsigned int getLength() {
127  return decSize;
128  }
