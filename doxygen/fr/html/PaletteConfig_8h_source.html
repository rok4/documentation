61 static void buildPLTE ( uint8_t pLTE[780],bool transparent, const uint8_t rgb[3] ) {
62  // Remplissage de la palette
63  pLTE[0] = 0;
64  pLTE[1] = 0;
65  pLTE[2] = 3;
66  pLTE[3] = 0;
67  pLTE[4] = 'P';
68  pLTE[5] = 'L';
69  pLTE[6] = 'T';
70  pLTE[7] = 'E';
71  if ( transparent ) {
72  for ( int i = 0; i < 256; i++ ) {
73  memcpy ( pLTE + 8 + 3*i, rgb, 3 );
74  }
75  } else for ( int i = 0; i < 256; i++ ) {
76  pLTE[3*i+8] = i + ( ( 255 - i ) *rgb[0] + 127 ) / 255;
77  pLTE[3*i+9] = i + ( ( 255 - i ) *rgb[1] + 127 ) / 255;
78  pLTE[3*i+10] = i + ( ( 255 - i ) *rgb[2] + 127 ) / 255;
79  }
80  uint32_t crc = crc32 ( 0, Z_NULL, 0 );
81  crc = crc32 ( crc, pLTE + 4, 3*256+4 );
82  * ( ( uint32_t* ) ( pLTE + 256*3 + 8 ) ) = bswap_32 ( crc );
83 }
