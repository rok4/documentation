577  static void add_slab_infos(std::string origin_slab_name, Context* data_context, std::string data_slab_name, int tiles_number, uint8_t* offsets, uint8_t* sizes) {
578  // On utilise le nom original de la dalle (celle à lire a priori) comme clé dans la map
579  // Potentiellement ce nom est différent de la vraie dalle contenant la donnée (dans le cas d'une dalle symbolique)
580  // mais c'est via cette dalle symbolique que la donnée est a priori requêtée
581  // donc c'est ce nom qu'on utilisera pour l'interrogation du cache
582 
583  IndexElement* elem = new IndexElement(origin_slab_name, data_context, data_slab_name, tiles_number, offsets, sizes);
584 
585  // L'information n'est a priori pas dans le cache car
586  // Soit elle était dans le cache et valide, alors on aurait utiliser ce cahe et on n'aurait pas fait appel à cet ajout
587  // Soit elle était dans le cache mais obsolète, alors on l'aurait déjà supprimé du cache
588 
589  if (cache.size() == size) {
590  // On réupcère le dernier élément du cache
591  IndexElement* last = cache.back();
592  // On le vire du cache
593  cache.pop_back();
594  // On le déréférence de la map d'accès
595  map.erase(last->key);
596  delete last;
597  }
598 
599  // update reference
600  cache.push_front(elem);
601  map[origin_slab_name] = cache.begin();
602  };

621  static bool get_slab_infos(std::string key, int tile_number, Context** data_context, std::string* data_slab_name, uint32_t* offset, uint32_t* size) {
622  std::unordered_map<std::string, std::list<IndexElement *>::iterator>::iterator it = map.find ( key );
623  if ( it == map.end() ) {
624  return false;
625  } else {
626  // Gestion de la péremption du cache (une heure max)
627  std::time_t now = std::time(NULL);
628  if (now - (*(it->second))->date > validity) {
629  delete *(it->second);
630  cache.erase(it->second);
631  map.erase(it);
632  return false;
633  }
634 
635  *data_slab_name = (*(it->second))->name;
636  *data_context = (*(it->second))->context;
637  *offset = (*(it->second))->offsets.at(tile_number);
638  *size = (*(it->second))->sizes.at(tile_number);
639 
640  // On fait le choix de ne pas remettre en tête du cache cet élément, même s'il vient d'être accéder
641  // C'est parce qu'en plus d'avoir une taille limite, les élément cachés ont une date de péromption
642  // Il serait donc dommage de remettre en avant (donc plus loin d'une suppression par taille de cache atteinte)
643  // un élément qui va de toute manière finir par être obsolète.
644  // C'est une différence avec une cache LRU classique
645 
646  return true;
647  }
648  };
