291  static Context * addContext(ContextType::eContextType type,std::string tray) {
292  Context* ctx;
293  std::pair<ContextType::eContextType,std::string> key = make_pair(type,tray);
294  BOOST_LOG_TRIVIAL(debug) << "On essaye d'ajouter la clé " << ContextType::toString(key.first) <<" / " << key.second ;
295 
296  std::map<std::pair<ContextType::eContextType,std::string>, Context*>::iterator it = pool.find (key);
297  if ( it != pool.end() ) {
298  //le contenant est déjà existant et donc connecté
299  return it->second;
300 
301  } else {
302  // ce contenant n'est pas encore connecté, on va créer la connexion
303  // on créé le context selon le type de stockage
304  switch(type){
305 #if OBJECT_ENABLED
306  case ContextType::SWIFTCONTEXT:
307  ctx = new SwiftContext(tray);
308  break;
309  case ContextType::CEPHCONTEXT:
310  ctx = new CephPoolContext(tray);
311  break;
312  case ContextType::S3CONTEXT:
313  ctx = new S3Context(tray);
314  break;
315 #endif
316  case ContextType::FILECONTEXT:
317  ctx = new FileContext(tray);
318  break;
319  default:
320  //ERREUR
321  BOOST_LOG_TRIVIAL(error) << "Ce type de contexte n'est pas géré.";
322  return NULL;
323  }
324 
325  // on connecte pour vérifier que ce contexte est valide
326  if (!(ctx->connection())) {
327  BOOST_LOG_TRIVIAL(error) << "Impossible de connecter au contexte de type " << ContextType::toString(type) << ", contenant " << tray;
328  delete ctx;
329  return NULL;
330  }

332 
333  //BOOST_LOG_TRIVIAL(debug) << "On insère ce contexte " << ctx->toString() ;

576  static void add_slab_infos(std::string origin_slab_name, Context* data_context, std::string data_slab_name, int tiles_number, uint8_t* offsets, uint8_t* sizes) {
577  // On utilise le nom original de la dalle (celle à lire a priori) comme clé dans la map
578  // Potentiellement ce nom est différent de la vraie dalle contenant la donnée (dans le cas d'une dalle symbolique)
579  // mais c'est via cette dalle symbolique que la donnée est a priori requêtée
580  // donc c'est ce nom qu'on utilisera pour l'interrogation du cache
581 
582  CacheElement* elem = new CacheElement(origin_slab_name, data_context, data_slab_name, tiles_number, offsets, sizes);
583 
584  // L'information n'est a priori pas dans le cache car
585  // Soit elle était dans le cache et valide, alors on aurait utiliser ce cahe et on n'aurait pas fait appel à cet ajout
586  // Soit elle était dans le cache mais obsolète, alors on l'aurait déjà supprimé du cache
587 
588  if (cache.size() == size) {
589  // On réupcère le dernier élément du cache
590  CacheElement* last = cache.back();
591  // On le vire du cache
592  cache.pop_back();
593  // On le déréférence de la map d'accès
594  map.erase(last->key);
595  delete last;
596  }
597 
598  // update reference
599  cache.push_front(elem);
600  map[origin_slab_name] = cache.begin();
601  };

620  static bool get_slab_infos(std::string key, int tile_number, Context** data_context, std::string* data_slab_name, uint32_t* offset, uint32_t* size) {
621  std::unordered_map<std::string, std::list<CacheElement *>::iterator>::iterator it = map.find ( key );
622  if ( it == map.end() ) {
623  return false;
624  } else {
625  // Gestion de la péremption du cache (une heure max)
626  std::time_t now = std::time(NULL);
627  if (now - (*(it->second))->date > validity) {
628  delete *(it->second);
629  cache.erase(it->second);
630  map.erase(it);
631  return false;
632  }
633 
634  *data_slab_name = (*(it->second))->name;
635  *data_context = (*(it->second))->context;
636  *offset = (*(it->second))->offsets.at(tile_number);
637  *size = (*(it->second))->sizes.at(tile_number);
638 
639  // On fait le choix de ne pas remettre en tête du cache cet élément, même s'il vient d'être accéder
640  // C'est parce qu'en plus d'avoir une taille limite, les élément cachés ont une date de péromption
641  // Il serait donc dommage de remettre en avant (donc plus loin d'une suppression par taille de cache atteinte)
642  // un élément qui va de toute manière finir par être obsolète.
643  // C'est une différence avec une cache LRU classique
644 
645  return true;
646  }
647  };
