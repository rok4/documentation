549  bool isCompatibleWith ( Image* pImage ) {
550 
551  if ( crs != NULL && crs->isDefine() && pImage->getCRS() != NULL && pImage->getCRS()->isDefine() && ! crs->cmpRequestCode(pImage->getCRS()->getRequestCode()) ) {
552  BOOST_LOG_TRIVIAL(debug) << "Different CRS" ;
553  return false;
554  }
555 
556  if ( getChannels() != pImage->getChannels() ) {
557  BOOST_LOG_TRIVIAL(debug) << "Different channels" ;
558  return false;
559  }
560 
561  double epsilon_x=std::min ( getResX(), pImage->getResX() ) /1000.;
562  double epsilon_y=std::min ( getResY(), pImage->getResY() ) /1000.;
563 
564  if ( fabs ( getResX()-pImage->getResX() ) > epsilon_x ) {
565  BOOST_LOG_TRIVIAL(debug) << "Different X resolutions" ;
566  return false;
567  }
568  if ( fabs ( getResY()-pImage->getResY() ) > epsilon_y ) {
569  BOOST_LOG_TRIVIAL(debug) << "Different Y resolutions" ;
570  return false;
571  }
572 
573  if ( fabs ( getPhaseX()-pImage->getPhaseX() ) > 0.001 && fabs ( getPhaseX()-pImage->getPhaseX() ) < 0.999 ) {
574  BOOST_LOG_TRIVIAL(debug) << "Different X phasis : " << getPhaseX() << " and " << pImage->getPhaseX() ;
575  return false;
576  }
577  if ( fabs ( getPhaseY()-pImage->getPhaseY() ) > 0.001 && fabs ( getPhaseY()-pImage->getPhaseY() ) < 0.999 ) {
578  BOOST_LOG_TRIVIAL(debug) << "Different Y phasis : " << getPhaseY() << " and " << pImage->getPhaseY() ;
579  return false;
580  }
581 
582  return true;
583  }
