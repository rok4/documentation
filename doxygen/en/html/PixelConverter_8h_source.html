216  T defaultAlpha;
217  if (sizeof(T) == 1) {
218  defaultAlpha = (T) 255;
219  } else if (sizeof(T) == 2) {
220  defaultAlpha = (T) 65535;
221  } else {
222  defaultAlpha = (T) 1;
223  }
224 
225  /********************** Depuis 1 canal ********************/
226  if ( inSamplesPerPixel == 1) {
227  if (outSamplesPerPixel == 2) {
228  for ( int i = 0; i < width; i++ ) {
229  bufferto[2*i] = bufferfrom[i];
230  bufferto[2*i+1] = defaultAlpha;
231  }
232  return;

253  bufferto[i] = bufferfrom[2*i];
254  }
