172  double y = sourceImages[i]->l2y ( 0 );
173 
174  rowsOffsets.push_back(y2l ( y ));
175  c0s.push_back(std::max ( 0,x2c ( sourceImages[i]->getXmin() + 0.5*sourceImages[i]->getResX() ) ));
176  c1s.push_back(std::min ( width - 1,x2c ( sourceImages[i]->getXmax() - 0.5*sourceImages[i]->getResX() ) ));
177  c2s.push_back(std::max ( 0, sourceImages[i]->x2c ( bbox.xmin + 0.5*resx ) ) );
