293 template<typename T>
294 TiffNodataManager<T>::TiffNodataManager ( uint16 channels, int* tv, bool touchEdges, int* dv, int* nv, int t ) :
295  maxChannels ( channels ), touchEdges ( touchEdges ), tolerance ( t ) {
296 
297  targetValue = new T[channels];
298  dataValue = new T[channels];
299  nodataValue = new T[channels];
300 
301  for ( int i = 0; i < channels; i++ ) {
302  targetValue[i] = ( T ) tv[i];
303  dataValue[i] = ( T ) dv[i];
304  nodataValue[i] = ( T ) nv[i];
305  }
306 
307  if ( memcmp ( tv,nv,channels*sizeof ( int ) ) ) {
308  newNodataValue = true;
309  } else {
310  // La nouvelle valeur de non-donnée est la même que la couleur cible : on ne change pas la couleur de non-donnée
311  newNodataValue = false;
312  }
313 
314  if ( touchEdges && memcmp ( tv,dv,channels*sizeof ( int ) ) ) {
315  // Pour changer la couleur des données contenant la couleur cible, il faut avoir l'option "touche les bords".
316  // Sinon, par définition, aucun pixel de la couleur cible est à considérer comme de la donnée.
317  removeTargetValue = true;
318  } else {
319  // La nouvelle valeur de donnée est la même que la couleur cible : on ne supprime donc pas la couleur cible des données
320  removeTargetValue = false;
321  }
322 }

325 bool TiffNodataManager<T>::treatNodata ( char* inputImage, char* outputImage, char* outputMask ) {
326  if ( ! newNodataValue && ! removeTargetValue && ! outputMask ) {
327  BOOST_LOG_TRIVIAL(info) << "Have nothing to do !" ;
328  return true;
329  }
330 
331  FileImageFactory FIF;
332  FileImage* sourceImage = FIF.createImageToRead(inputImage);
333 
334  if ( sourceImage == NULL ) {
335  BOOST_LOG_TRIVIAL(error) << "Cannot create the input image "<< inputImage ;
336  return false;
337  }
338 
339  /* On mémorise certaines informations sur l'image en cours de traitement */
340  width = sourceImage->getWidth();
341  height = sourceImage->getHeight();
342 
343  int bitspersample = sourceImage->getBitsPerSample();
344  Photometric::ePhotometric photometric = sourceImage->getPhotometric();
345  Compression::eCompression compression = sourceImage->getCompression();
346  SampleFormat::eSampleFormat sampleformat = sourceImage->getSampleFormat();
347  samplesperpixel = sourceImage->getChannels();
348 
349  if ( samplesperpixel > maxChannels ) {
350  BOOST_LOG_TRIVIAL(error) << "The nodata manager is not adapted (samplesperpixel have to be " << maxChannels <<
351  " or less) for the image " << inputImage << " (" << samplesperpixel << ")";
352  return false;
353  }
354 
355  /*************** Chargement de l'image ***************/
356 
357  T *IM = (T*) malloc (width * height * samplesperpixel * sizeof(T));
358  if ( IM == NULL ) {
359  BOOST_LOG_TRIVIAL(error) << "Cannot allocate a buffer of size " << width * height * samplesperpixel * sizeof(T) << " bytes" ;
360  return false;
361  }
362 
363  BOOST_LOG_TRIVIAL(debug) << "We load input image into memory : " << width * height * samplesperpixel * sizeof(T) / 1024 << " kilobytes";
364  for ( int h = 0; h < height; h++ ) {
365  if (sourceImage->getline(IM + width*samplesperpixel*h, h) == 0) {
366  BOOST_LOG_TRIVIAL(error) << "Cannot read line " << h ;
367  return false;
368  }
369  }
370 
371  delete sourceImage;
372 
373  BOOST_LOG_TRIVIAL(debug) << "Premier pixel : " << (int)IM[0] << "," << (int)IM[1] << "," << (int)IM[2];
374 
375  /************* Calcul du masque de données ***********/
376 
377  uint8_t *MSK = new uint8_t[width * height];
378 
379  bool containNodata = identifyNodataPixels ( IM, MSK );
380 
381  /*************** Modification des pixels *************/
382 
383  if ( removeTargetValue ) {
384  BOOST_LOG_TRIVIAL(debug) << "The 'targetValue' data pixels are replaced by 'dataValue' pixels";
385  changeDataValue ( IM, MSK );
386  }
387 
388  if ( newNodataValue ) {
389  BOOST_LOG_TRIVIAL(debug) << "Nodata pixels which touch edges are replaced by 'nodataValue' pixels";
390  changeNodataValue ( IM, MSK );
391  }
392 
393  /**************** Ecriture de l'images ****************/
394 
395  /* Seulement si on a modifié l'image. Si seule l'écriture du masque nous intéressait, on ne réecrit pas l'image,
396  * même si un chemin d'image différent est fourni pour la sortie */
397  if ( removeTargetValue || newNodataValue ) {
398 
399  FileImage* destImage = FIF.createImageToWrite(
400  outputImage, BoundingBox<double>(0,0,0,0), -1, -1, width, height,
401  samplesperpixel, sampleformat, bitspersample, photometric, compression
402  );
403 
404  if ( destImage == NULL ) {
405  BOOST_LOG_TRIVIAL(error) << "Cannot create the output image "<< outputImage ;
406  return false;
407  }
408 
409  BOOST_LOG_TRIVIAL(debug) << "We write the output image " << outputImage;
410  destImage->writeImage(IM);
411 
412  delete destImage;
413 
414  } else {
415  if ( memcmp ( inputImage, outputImage, sizeof ( outputImage ) ) )
416  BOOST_LOG_TRIVIAL(info) << "The image have not be modified, the file '" << outputImage <<"' is not written" ;
417  }
418 
419  /**************** Ecriture du masque ? ****************/
420  if ( outputMask ) {
421  if (containNodata) {
422  FileImage* destMask = FIF.createImageToWrite(
423  outputMask, BoundingBox<double>(0,0,0,0), -1, -1, width, height,
424  1, SampleFormat::UINT, 8, Photometric::MASK, Compression::DEFLATE
425  );
426 
427  if ( destMask == NULL ) {
428  BOOST_LOG_TRIVIAL(error) << "Cannot create the output mask "<< outputMask ;
429  return false;
430  }
431 
432  BOOST_LOG_TRIVIAL(debug) << "We write the output mask " << outputMask;
433  destMask->writeImage(MSK);
434 
435  delete destMask;
436  } else {
437  BOOST_LOG_TRIVIAL(info) << "The image contains only data, the mask '" << outputMask <<"' is not written" ;
438  }
439  }
440 
441  free(IM);
442  delete[] MSK;
443 
444  return true;

451  for ( int i = 0; i < samplesperpixel; i++ ) {
452  pixint = ( int ) pix[i];

461 template<typename T>
462 bool TiffNodataManager<T>::identifyNodataPixels ( T* IM, uint8_t* MSK ) {
463 
464  BOOST_LOG_TRIVIAL(debug) << "Identify nodata pixels..." ;
465  memset ( MSK, 255, width * height );
466 
467  bool containNodata = false;
468 
469  if ( touchEdges ) {
470  BOOST_LOG_TRIVIAL(debug) << "\t...which touch edges" ;
471  // On utilise la couleur targetValue et on part des bords
472  queue<unsigned long long> Q;
473 
474  // Initialisation : we identify front pixels which are 'targetValue'
475  for ( unsigned long long pos = 0; pos < width; pos++ ) { // top
476  if ( isTargetValue ( IM + samplesperpixel * pos ) ) {
477  Q.push ( pos );
478  MSK[pos] = 0;
479  }
480  }
481  for ( unsigned long long pos = width* ( height-1 ); pos < width*height; pos++ ) { // bottom
482  if ( isTargetValue ( IM + samplesperpixel * pos ) ) {
483  Q.push ( pos );
484  MSK[pos] = 0;
485  }
486  }
487  for ( unsigned long long pos = 0; pos < width*height; pos += width ) { // left
488  if ( isTargetValue ( IM + samplesperpixel * pos ) ) {
489  Q.push ( pos );
490  MSK[pos] = 0;
491  }
492  }
493  for ( unsigned long long pos = width -1; pos < width*height; pos+= width ) { // right
494  if ( isTargetValue ( IM + samplesperpixel * pos ) ) {
495  Q.push ( pos );
496  MSK[pos] = 0;
497  }
498  }
499 
500  if ( Q.empty() ) {
501  // No nodata pixel identified, nothing to do
502  return false;
503  }
504 
505  containNodata = true;
506 
507  // while there are 'targetValue' pixels which can propagate, we do it
508  while ( !Q.empty() ) {
509  unsigned long long pos = Q.front();
510  Q.pop();
511  unsigned long long newpos;
512  if ( pos % width > 0 ) {
513  newpos = pos - 1;
514  if ( MSK[newpos] && isTargetValue ( IM + newpos*samplesperpixel ) ) {
515  MSK[newpos] = 0;
516  Q.push ( newpos );
517  }
518  }
519  if ( pos % width < width - 1 ) {
520  newpos = pos + 1;
521  if ( MSK[newpos] && isTargetValue ( IM + newpos*samplesperpixel ) ) {
522  MSK[newpos] = 0;
523  Q.push ( newpos );
524  }
525  }
526  if ( pos / width > 0 ) {
527  newpos = pos - width;
528  if ( MSK[newpos] && isTargetValue ( IM + newpos*samplesperpixel ) ) {
529  MSK[newpos] = 0;
530  Q.push ( newpos );
531  }
532  }
533  if ( pos / width < height - 1 ) {
534  newpos = pos + width;
535  if ( MSK[newpos] && isTargetValue ( IM + newpos*samplesperpixel ) ) {
536  MSK[newpos] = 0;
537  Q.push ( newpos );
538  }
539  }
540  }
541 
542  } else {
543  BOOST_LOG_TRIVIAL(debug) << "\t..., all pixels in 'target color'" ;
544  // Tous les pixels de la couleur targetValue sont à considérer comme du nodata
545 
546  for ( unsigned long long i = 0; i < width * height; i++ ) {
547 
548  if ( isTargetValue ( IM+i*samplesperpixel ) ) {
549  containNodata = true;
550  MSK[i] = 0;
551  }
552  }
553 
554  }
555 
556  return containNodata;
557 }
